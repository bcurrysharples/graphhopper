         private String destinationId;

             this.startTime = startTime;
             this.originId = originId;
             this.destinationId = destinationId;
             this.zones = zones;
         }

         }

             return startTime;
         }
 
         String getOriginId() {
             return originId;
         }
 
         String getDestinationId() {
             return destinationId;
         }
 
         Set<String> getZones() {
             return zones;
         }
 
     }
 
     public final List<Segment> segments = new ArrayList<>();
 
 
 }
